def calcularPercentualNotaIMDb(df, nome_plataforma, limiar=8):
    qtd = 0
    for index, row in df.iterrows():
        if(row['IMDb'] > 8):
            qtd+=1
    percentual = qtd/len(df)*100
    print('\t-', nome_plataforma, "= {:.2f}%".format(percentual))
    return percentual
